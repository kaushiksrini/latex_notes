\begin{gather*}
	\text{Maximise } \quad 1 + x_1^2(x_2-1)^3 e^{-x_1-x_2}\\ 
	x_1 + x_2 \leq 6\\
	x_2 > \log x_1 \\
	x_1 \geq 0\\
	x_2 \geq 0
\end{gather*}

\begin{definition-N}
	We say $x^*$ is an \underline{optimal solution} if
	\begin{itemize}
		\item $x^* \in \mathbb{S}$
		\item For any $y \in \mathbb{S}$
	\end{itemize}
\end{definition-N}
